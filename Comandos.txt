pwd - Mostrar o caminho onde o terminal está sendo rodado
ls - listar arquivos e diretórios
echo - exibir mensagem
cat - ler aqrquivo no terminal
clear - limpar o terminal
ls -l - mostra as informações do diretório atual
ls -la - mostra as informações(inclusive arquivos ocultos) do diretório atual
man - mostra o manual (doc) do comando
cd - acessar diretórios
cd .. - voltar diretórios
mkdir - criar diretórios
rmdir - apagar diretórios vazios
rm - apagar arquivos
rm -r - apagar arquivos e diretórios
